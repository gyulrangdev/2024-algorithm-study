function findMinHeightTrees(n: number, edges: number[][]): number[] {
  if (n === 1) return [0];
  if (n === 2) return [0, 1];

  // 각 노드의 이웃 노드들을 저장할 맵 생성
  const adj = new Map<number, Set<number>>();
  for (let i = 0; i < n; i++) {
    adj.set(i, new Set());
  }

  // 그래프 구성
  for (const [a, b] of edges) {
    adj.get(a)!.add(b);
    adj.get(b)!.add(a);
  }

  // 리프 노드들 찾기
  let leaves: number[] = [];
  for (let i = 0; i < n; i++) {
    if (adj.get(i)!.size === 1) {
      leaves.push(i);
    }
  }

  // 남은 노드가 2개 이하가 될 때까지 리프 노드들 제거
  let remainingNodes = n;
  while (remainingNodes > 2) {
    remainingNodes -= leaves.length;
    const newLeaves: number[] = [];

    for (const leaf of leaves) {
      const neighbor = Array.from(adj.get(leaf)!)[0];
      adj.get(neighbor)!.delete(leaf);
      if (adj.get(neighbor)!.size === 1) {
        newLeaves.push(neighbor);
      }
    }

    leaves = newLeaves;
  }

  return leaves;
}
